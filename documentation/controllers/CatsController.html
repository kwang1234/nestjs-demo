import {
  Controller,
  Get,
  Req,
  Post,
  HttpCode,
  Header,
  Param,
  Body,
  Query,
  HttpException,
  HttpStatus,
  UnauthorizedException,
  UseInterceptors,
  UseGuards,
  UsePipes,
  ValidationPipe,
} from '@nestjs/common';
import { Request } from 'express';
import { CreateCatDto } from './create-cat.dto';
import { CatsService } from './cat.service';
import { CatsLoggingInterceptor } from './logging.interceptor';
import { CatsAuthGuard } from './cat.guard';
import { ApiResponse } from '@nestjs/swagger';

@Controller('Cats')
export class CatsController {
  constructor(private readonly catsService: CatsService) {}

  @Get()
  @ApiResponse({
    status: 201,
    description: 'The record has been successfully created.',
  })
  async findAll(@Query() query: any) {
    return { name: 'sb' };
  }

  @Post()
  @UseInterceptors(CatsLoggingInterceptor)
  @UseGuards(CatsAuthGuard)
  @UsePipes(new ValidationPipe())
  CreateCat(@Body() createCatDto: CreateCatDto) {
    this.catsService.create(createCatDto);
    return this.catsService.findAll();
  }

  @Post()
  createCall(@Body() createCatDto: CreateCatDto): string {
    return createCatDto.name;
  }

  @Post('ab*d')
  @HttpCode(204)
  @Header('Cache-control', 'none')
  getCat(@Req() req: Request) {
    const cat = req.body;
    return cat.name;
  }

  @Get(':id')
  findOne(@Param() params): string {
    return `this action return a ${params.id}`;
  }
}
